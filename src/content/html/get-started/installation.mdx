import _components from "@content/_components"
import CodeSnippet from "@components/utilities/CodeSnippet"
import BoilerplateCard from "@components/utilities/BoilerplateCard"
import csscode from "@lib/csscode"

export const components = {
    ..._components
}

export const configCode = `import { palettes } from "@tailus/themer-plugins"

module.exports = {
    content: [
        './index.html',
        "./src/**/*.{html,js}"
    ],
    theme: {
        extend: {
            colors : palettes.trust
        },
    }
};`

export const config = {
    code: configCode,
    lang: "ts",
    highlightedLines:"1, 10",
}
